
# ENCUENTRE TODOS LOS ARCHIVOS EN EL DIRECTORIO /USR/BIN (RECURSIVO) QUE SE HAYAN MODIFICADO EN LAS ULTIMAS 2 HORAS
find /usr/bin -type f -mmin -120

# BUSCAR TODOS LOS ARCHIVOS Q LOS QUE NO SE ACCEDIÃ“ EN LOS ULTIMOS 10 DIAS CON LA EXTENSION .PNG
find / -type f -atime +10 -name "*.png"

# ENCUENTRE TODOS LOS ARCHIVOS EN EL DIRECTORIO /USR/BIN (RECURSIVO) QUE SON PROPIEDAD DE ROOT Y TIENEN EL PERMISO SUID
find /usr/bin -type f -user root -perm -u=s

# BUSQUE TODOS LOS ARCHIVOS CON PERMISO DE ESCRITURA PARA EL GRUPO "OTROS", INDEPENDIENTEMENTE DE CUALQUIER OTRO PERMISO, CON LA EXTENSION ".SH" 
find / -type f -perm -o=w -name "*.sh"

# ENCUENTRE TODOS LOS ARCHIVOS QUE SOLO CUALQUIER PERSONA PUEDA LEER (FORMATO OCTAL)
find / -type f -perm /444

# ENCUENTRE TODOS LOS ARCHIVOS QUE EL PROPIETARIO PUEDA LEER Y ESCRIBIR EXACTAMENTE, Y QUE TODOS LOS DEMAS PUEDAN LEERLO
find / -type f -perm 644
